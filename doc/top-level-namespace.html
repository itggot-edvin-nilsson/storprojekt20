def clearExpiredTokens()
  Thread.new do
    while true
      sleep(5 * 60)
      $tokens.reject! {|x, y| y[0] < Time.now}
    end
  end
end

post '/login' do
  response = login(params[:username], params[:password])
  if response.successful
    session[:token] = response.data
    if session[:redirect]
      path = session[:redirect]
      session[:redirect] = nil
      redirect(path)
    else
      redirect('/')
    end
  else
    session[:error] = response.data
    redirect('/login')
  end
end
